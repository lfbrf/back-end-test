Definições:


O intuito dessa página é apenas facilitar as requisições da API, sem considerar o layout da mesma
Por gentileza disparar todas as requisiçõe atravé dessa URL (por já conter o jquery)
Detalhe: considerando porta 8080, se usar outra é necessário ajustar nas requisições AJAX o campo url
Títulos Marcados por (*) são os requisitos obrigatórios

Selecione a opção desejada para verificar a documentação da API

Listar todos os produtos


http://localhost:8080/api/produtos/listAll

Listar produto por id, considerando id 1

http://localhost:8080/api/produtos/1

Adicionar um novo produto:


var produto = {"id":1,"grupoProduto":{"id":1,"nome":"Grupo Um"},"valorUnitario":100.0,"nome":"Produto Novo"};
$.ajax({
data: JSON.stringify(produto),
contentType : 'application/json; charset=utf-8',
url : "http://localhost:8080/api/produtos/",
type : 'POST',
success : function(data) {
console.log("Produto Cadastrado com sucesso");
},
error : function(e) {
alert("Erro ao salvar produto, por favor verifique as informações e tente novamente");
alert(e.responseJSON.message);
er = e;
console.log("error: " + e.responseJSON.message);
} });

Editar um produto


var produto = {"id":1,"grupoProduto":{"id":1,"nome":"Grupo Um"},"valorUnitario":1000.0,"nome":"Produto Novo"};
var idGrupo = 1;
$.ajax({
data: JSON.stringify(produto),
contentType : 'application/json; charset=utf-8',
url : "http://localhost:8080/api/produtos/"+ idGrupo,
type : 'PUT',
success : function(data) {
console.log("Produto editado com sucesso");
console.log(data);
},
error : function(e) {
console.log("Erro ao editar produto, por favor verifique as informações e tente novamente");
er = e;
console.log("error: " + e.responseJSON.message);}});
